\newcommand*{\twopagepicture}[3]{%
    \checkoddpage
    \ifoddpage
        \expandafter\@firstofone
    \else
        \expandafter\afterpage
    \fi
    {\afterpage{%
    \if #1t%
        \thispagestyle{empty}%
        \afterpage{\thispagestyle{empty}}%
    \fi
    \begin{figure*}[#1]
        \if #1t%
            \vspace*{-\dimexpr1in+\voffset+\topmargin+\headheight+\headsep+#2\relax}%
        \fi
        \makebox[\textwidth][l]{%
        \relax\let\mywidth\paperwidth
            \hskip-\dimexpr1in+\hoffset+\evensidemargin\relax

        \adjustbox{trim=0 0 {.5\width} 0,clip}{\includegraphics[width=2\mywidth]{#3}}}%
        \if #1b%
            \vspace*{-\dimexpr\paperheight-\textheight-1in-\voffset-\topmargin-\headheight-\headsep+#2\relax}%
        \fi
    \end{figure*}%
    \begin{figure*}[#1]
        \if #1t%
            \vspace*{-\dimexpr1in+\voffset+\topmargin+\headheight+\headsep+#2\relax}%
        \fi
        \makebox[\textwidth][l]{%
        \let\mywidth\paperwidth
        \hskip-\dimexpr1in+\hoffset+\oddsidemargin\relax
        \adjustbox{trim={.5\width} 0 0 0,clip}{\includegraphics[width=2\mywidth]{#3}}}%
    	\if #1b%
            \vspace*{-\dimexpr\paperheight-\textheight-1in-\voffset-\topmargin-\headheight-\headsep+#2\relax}%
        \fi
    \end{figure*}%
    }}%
}
\makeatother
